// -------- Voucher Number -----------
function generateVoucherNumber(){
let last=localStorage.getItem("lastVoucherNumber");
if(!last){ last=60; }
else{ last=Number(last)+1; }
localStorage.setItem("lastVoucherNumber",last);
return "V-"+last;
}

// -------- Table & Row Functions --------
var table=document.getElementById("table");

function addRow(desc="",amount=""){
var row=table.insertRow();
row.insertCell(0).innerHTML=table.rows.length-1;
row.insertCell(1).innerHTML="<input value='"+desc+"'>";
row.insertCell(2).innerHTML="<input type='number' value='"+amount+"' oninput='calc()'>";
}

function removeLastRow(){
if(table.rows.length>2){
table.deleteRow(-1);
calc();
}
}

function clearTable(){
while(table.rows.length>1){
table.deleteRow(1);
}
}

function calc(){
var inputs=table.querySelectorAll("input[type=number]");
var total=0;
inputs.forEach(i=> total+=Number(i.value));
document.getElementById("total").innerText=total;
document.getElementById("totalWords").innerText=numberToBangla(total)+" টাকা মাত্র";
}

// -------- Save Voucher --------
function saveVoucher(){
var voucher=document.getElementById("voucher").value;
var rows=table.querySelectorAll("tr");

let items=[];
let hasValid=false;

for(var i=1;i<rows.length;i++){
var desc=rows[i].cells[1].querySelector("input").value.trim();
var amount=rows[i].cells[2].querySelector("input").value.trim();

if(desc!=="" && amount!==""){
hasValid=true;
items.push({desc,amount});
}
}

if(!hasValid){
alert("Description এবং Amount খালি থাকলে Save হবে না!");
return false;
}

var data={
purpose:purpose.value,
mgl:mgl.value,
approval:approvalBy.value,
receiver:receiverBy.value,
items:items
};

localStorage.setItem(voucher,JSON.stringify(data));
alert("Saved Successfully!");
return true;
}

// -------- Print Voucher --------
function printVoucher(){
if(saveVoucher()){
window.print();
newVoucher();
}
}

// -------- Search Voucher --------
function searchVoucher(){
var voucher=prompt("Enter Voucher No:");
if(!voucher) return;

var data=localStorage.getItem(voucher);
if(!data){ alert("Not Found!"); return; }

data=JSON.parse(data);

clearTable();

document.getElementById("voucher").value=voucher;
purpose.value=data.purpose;
mgl.value=data.mgl;
approvalBy.value=data.approval;
receiverBy.value=data.receiver;

data.items.forEach(i=> addRow(i.desc,i.amount));

calc();
}

// -------- New Voucher --------
function newVoucher(){
document.querySelectorAll("input").forEach(i=>{
if(!i.hasAttribute("readonly")) i.value="";
});

clearTable();

document.getElementById("voucher").value=generateVoucherNumber();

var d=new Date();
document.getElementById("date").value=d.getDate()+"-"+(d.getMonth()+1)+"-"+d.getFullYear();

addRow();
calc();
}

// -------- Init after Login --------
function initVoucher(){
document.getElementById("voucher").value=generateVoucherNumber();
var d=new Date();
document.getElementById("date").value=d.getDate()+"-"+(d.getMonth()+1)+"-"+d.getFullYear();
addRow();
}
